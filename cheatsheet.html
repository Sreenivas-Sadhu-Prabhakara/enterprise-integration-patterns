Group 01

Route basics

RouteBuilder.java
from("direct:orders")
  .routeId("order-intake")
  .log("received ${body}")
  .to("bean:validate")
  .to("seda:accepted");

// a timer-driven source
from("timer:tick?period=5000")
  .routeId("feed")
  .to("direct:orders");

Group 02

Simple language

Simple expressions
${body}              // the message body
${header.country}    // one header (also headers.country)
${body.total}        // bean property → getTotal()
${exchangeProperty.count}

// predicates (use inside simple("..."))
${header.country} == 'US'
${body.total} > 100
${header.sku} regex '[A-Z]{3}-\d+'
${body} contains 'urgent'

Group 03

Routing EIPs

Routing DSL
.choice()
  .when(simple("${header.country} == 'US'")).to("direct:us")
  .when(simple("${header.country} == 'UK'")).to("direct:uk")
  .otherwise().to("direct:intl")
.end()

.filter(simple("${body.total} > 0")).to("direct:billable")

.split(body()).to("direct:line")     // one → many
.split(jsonpath("$.items"))
.split(body().tokenize(","))         // split a CSV body

.aggregate(header("orderId"),
      new GroupedBodyAggregationStrategy())
    .completionSize(10).completionTimeout(2000)

.recipientList(header("recipients")) // comma-separated URIs
.multicast().parallelProcessing()
    .to("direct:a", "direct:b").end()
.wireTap("direct:audit")             // fire-and-forget copy
.dynamicRouter(method(Router.class, "next"))
.routingSlip(header("itinerary"))

Group 04

Transformation

Transform & marshal
.transform(simple("Hello ${body.name}"))
.setBody(simple("${body.customer}"))
.setHeader("region", simple("${body.country}"))

// JSON to/from POJO (Jackson)
.unmarshal().json(JsonLibrary.Jackson, Order.class)
.marshal().json(JsonLibrary.Jackson)

// call a bean
.bean(OrderService.class, "enrich")
.to("bean:orderService?method=score")

// content enricher — merge in extra data
.enrich("direct:lookupCustomer", new MergeStrategy())
.pollEnrich("file:rates?fileName=fx.json")

Group 05

Reliability

Error handling
// global dead-letter + redelivery / backoff
errorHandler(deadLetterChannel("seda:dead")
    .maximumRedeliveries(3)
    .redeliveryDelay(1000)
    .backOffMultiplier(2)
    .useOriginalMessage());

// per-exception policy
onException(IOException.class)
    .maximumRedeliveries(2)
    .handled(true)
    .to("direct:failed");

// process each id at most once
from("direct:in")
  .idempotentConsumer(header("orderId"),
      MemoryIdempotentRepository
          .memoryIdempotentRepository(500))
  .to("direct:process");

Group 06

Endpoints

Endpoint URIs
direct:name    // sync · same thread · same JVM
seda:name      // async in-memory queue (staged)
vm:name        // async · across contexts in one JVM

jms:queue:orders     // point-to-point (one consumer)
jms:topic:events     // publish-subscribe (all subscribers)
kafka:orders?brokers=localhost:9092

timer:tick?period=5000
file:inbox?move=.done

Group 07

Testing

RouteTest.java
@CamelSpringBootTest
@SpringBootTest
class RouteTest {
  @Autowired CamelContext context;
  @Autowired ProducerTemplate template;
  @EndpointInject("mock:result") MockEndpoint mock;

  @Test void routesToResult() throws Exception {
    mock.expectedMessageCount(1);
    mock.expectedBodiesReceived("OK");
    template.sendBody("direct:in", "ping");
    mock.assertIsSatisfied();
  }
}

// swap a real endpoint for a mock at test time
AdviceWith.adviceWith(context, "order-intake",
    a -> a.weaveById("send").replace().to("mock:result"));

// assert message flow without mocks
NotifyBuilder done = new NotifyBuilder(context)
    .whenDone(1).create();
